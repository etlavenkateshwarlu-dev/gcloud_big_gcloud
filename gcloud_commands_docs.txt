Create a Compute Engine VM using gcloud
----------------------------------------
# Make sure correct project is set
gcloud config set project vctbatchnew45

# (Optional) Set default region & zone
gcloud config set compute/region us-central1
gcloud config set compute/zone us-central1-a


# Create a Compute Engine VM
gcloud compute instances create batch45 --zone=us-central1-a --machine-type=e2-medium --image-family=debian-12 --image-project=debian-cloud  --boot-disk-size=20GB
